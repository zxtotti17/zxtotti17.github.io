魔域资料库目录结构生成

2022-09-13
#游戏

魔域资料库类似于魔域的成就达成系统
成就的目录及内容通过后台配置
表结构设计上，保证一个主菜单id 和关联父菜单iD，同样作品下也是，支持拖拽排序，将完整的菜单构建成菜单树的结构，用父关联替代一个层级一个字段方便以后的拓展性不用加层级字段

CREATE TABLE `moyu_lists` (
  `id` char(24) NOT NULL COMMENT '编号',
   `menu_root_id` varchar(50) DEFAULT '' COMMENT '主菜单Id',
  `menu_prent_id`  varchar(50) DEFAULT '' COMMENT '关联父Id',
  `title` varchar(1000) DEFAULT ''  COMMENT '标题',
  `icon` varchar(500) CHARACTER SET utf8mb4 COLLATE  utf8mb4_general_ci NULL DEFAULT '' COMMENT '图标地址',
  `menu_level` int(11)  DEFAULT 1 COMMENT '目录层级',
  `sort_index` int(11) NULL DEFAULT 0 COMMENT  '排序号',
  `show_all` int(11) NULL DEFAULT 0 COMMENT '是否显示全部',
  `is_delete`  int(11) DEFAULT 0 COMMENT '是否删除',
  `create_time` datetime DEFAULT NULL  COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `menu_prent`  (`menu_root_id`,`menu_prent_id`) USING BTREE,
  KEY `level_sort`  (`menu_level`,`sort_index`) USING BTREE
) ENGINE=InnoDB DEFAULT  CHARSET=utf8mb4 ROW_FORMAT=DYNAMIC;
CREATE TABLE `moyu_works` (
  `id` char(24) NOT NULL COMMENT '编号',
   `menu_root_id` varchar(50) DEFAULT '' COMMENT '主菜单Id',
  `menu_id`  varchar(50) DEFAULT '' COMMENT '关联菜单Id',
  `works_id` varchar(50) DEFAULT ''  COMMENT '作品ID',
  `game_condition` int(11) DEFAULT 0 COMMENT '游戏解锁条件',
   `game_condition_desc` varchar(1000) DEFAULT '' COMMENT '游戏解锁条件描述',
   `works_title` varchar(1000) DEFAULT 0 COMMENT '作品标题',
  `item_id` int(11)  DEFAULT 0 COMMENT '游戏物品id',
  `menu_level` int(11) DEFAULT 1 COMMENT  '目录层级',
  `sort_index` int(11) NULL DEFAULT 0 COMMENT '排序号',
  `is_delete`  int(11) DEFAULT 0 COMMENT '是否删除',
  `create_time` datetime DEFAULT NULL  COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `menu_prent`  (`menu_root_id`,`menu_id`) USING BTREE,
  KEY `sort_index` (`sort_index`)  USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 ROW_FORMAT=DYNAMIC;

通过查询出的menu_level 和 sort_index的排序规则构建层级

private async  getMenuLists(doingList:MoyuLists[], isUse?:boolean):  Promise<any[]>{
        let result:any = [];
        let  indexMap:any = {};
        for(let i = 0, l = doingList.length; i < l;  i++){
            let tmp:MoyuLists = doingList[i];
             if(tmp.menu_level == 1){
                let oneMap:any =  {};
                if(isUse){
                    oneMap =  MoyuListsBody.buildOne(tmp);
                }else{
                     oneMap = new MoyuLists(tmp);
                    oneMap.childs =  [];
                }
                 result.push(oneMap);
                indexMap[tmp.id] = result.length -  1;
            }else if(tmp.menu_level == 2 &&  indexMap[tmp.menu_root_id] >= 0){
                let oneIndex =  indexMap[tmp.menu_root_id] || 0;
                let oneMap:any =  {};
                if(isUse){
                    oneMap =  MoyuListsBody.buildOne(tmp);
                }else{
                     oneMap = new MoyuLists(tmp);
                    oneMap.childs =  [];
                }
                 result[oneIndex].childs.push(oneMap);
                let twoL =  result[oneIndex].childs.length - 1;
                indexMap[tmp.id] =  [oneIndex, twoL];
            }else if(tmp.menu_level == 3 &&  indexMap[tmp.menu_prent_id]){
                let twoIndex =  indexMap[tmp.menu_prent_id];
                let oneMap:any =  {};
                if(isUse){
                    oneMap =  MoyuListsBody.buildOne(tmp);
                }else{
                     oneMap = new MoyuLists(tmp);
                    oneMap.childs =  [];
                }
                 result[twoIndex[0]].childs[twoIndex[1]].childs.push(oneMap);
                 let threeL = result[twoIndex[0]].childs[twoIndex[1]].childs.length -  1;
                indexMap[tmp.id] = [twoIndex[0], twoIndex[1],  threeL];
            }else if(tmp.menu_level == 4 &&  indexMap[tmp.menu_prent_id]){
                let threeIndex =  indexMap[tmp.menu_prent_id];
                let oneMap:any =  {};
                if(isUse){
                    oneMap =  MoyuListsBody.buildOne(tmp);
                }else{
                     oneMap = new MoyuLists(tmp);
                    oneMap.childs =  [];
                }
                 result[threeIndex[0]].childs[threeIndex[1]].childs[threeIndex[2]].childs.push(oneMap);
                 let fourL =  result[threeIndex[0]].childs[threeIndex[1]].childs[threeIndex[2]].childs.length  - 1;
                indexMap[tmp.id] = [threeIndex[0], threeIndex[1],  threeIndex[2], fourL];
            }
        }
        return  result;    
    }

以上是构建4级目录的树的方式，下标索引用一个map对象维护，一般4~5级的层级即可， 如果需要在拓展就继续加对应代码
后台更新和添加的时候都mq发布出来刷新指令，不去维护树中的改变，重新刷新缓存中的使用的菜单树，达到热更新的效果

public async refreshUseLists(): Promise<any> {
         EventManager.getInstance().execSync('moyuListsSync.refresh', "")
         return true;
    }
